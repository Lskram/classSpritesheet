        currAnim = player.state.Run

        currAnim = player.state.Run

        currAnim = player.state.Run

    elseif love.keyboard.isDown("down") then
        currAnim = player.state.Run

        currAnim = player.state.Attack

        currAnim = player.state.Idel

end
